---
title: "React adapter"
description: "@blinn-motion/react — declarative BlinnMotion component with a swappable renderer."
---

`@blinn-motion/react` wraps the engine in a component and a hook. Pick the paint backend with one
prop; drive playback through a ref or controlled `progress`.

## Install

```bash
npm i @blinn-motion/react
```

Peer: React ≥ 17.

## The component

```tsx
import { BlinnMotion } from "@blinn-motion/react";
import doc from "./card.motion.json";

export default function Hero() {
  return (
    <BlinnMotion
      doc={doc}
      renderer="canvas"   // or "dom"
      loop
      autoplay
    />
  );
}
```

## Props

<ParamField path="doc" type="MotionDoc" required>
  The animation document to play.
</ParamField>
<ParamField path="renderer" type='"dom" | "canvas"' default='"dom"'>
  Which adapter paints the resolved tree.
</ParamField>
<ParamField path="loop" type="boolean" default="true">
  Restart when the timeline ends.
</ParamField>
<ParamField path="autoplay" type="boolean" default="true">
  Start playing on mount. Forced off when `progress` is set.
</ParamField>
<ParamField path="rate" type="number" default="1">
  Playback speed multiplier.
</ParamField>
<ParamField path="progress" type="number">
  Controlled progress `0…1`. Ideal for scroll- / gesture-linked motion.
</ParamField>
<ParamField path="onFrame" type="(t: number, fraction: number) => void">
  Per-frame callback for scrubbers and timecode readouts.
</ParamField>

## Driving playback with a ref

```tsx
import { useRef } from "react";
import { BlinnMotion, type BlinnMotionHandle } from "@blinn-motion/react";

function Controlled({ doc }) {
  const ref = useRef<BlinnMotionHandle>(null);
  return (
    <>
      <BlinnMotion ref={ref} doc={doc} renderer="dom" />
      <button onClick={() => ref.current?.toggle()}>Play / pause</button>
      <input
        type="range" min={0} max={1} step={0.001}
        onChange={(e) => ref.current?.seekFraction(Number(e.target.value))}
      />
    </>
  );
}
```

Ref surface: `play` · `pause` · `stop` · `toggle` · `seek` · `seekFraction` · `setProgress` · `setRate` · `player`.

## Hook

```tsx
import { useRef } from "react";
import { useBlinnMotion } from "@blinn-motion/react";

const host = useRef<HTMLDivElement>(null);
const player = useBlinnMotion(host, doc, { renderer: "canvas" });
// <div ref={host} />
```

## Next.js

Use a **client** component — motion does not paint on the server:

```tsx
"use client";
import { BlinnMotion } from "@blinn-motion/react";
import doc from "./card.motion.json";

export function MotionHero() {
  return <BlinnMotion doc={doc} renderer="dom" loop autoplay />;
}
```
